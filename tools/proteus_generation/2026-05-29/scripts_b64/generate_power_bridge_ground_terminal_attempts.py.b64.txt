def donor_ground_bridge_by_power_layout(node_label: str) -> bytes:
    if len(node_label) != 2:
        raise ValueError('node label must be two characters')
    c = extract_donor_power_bridge_cluster()
    c = c.replace(b'\xff\x02N1', b'\xff\x02' + node_label.encode('ascii'), 1)
    if b'\x09$TERPOWER' not in c:
        raise RuntimeError('length-prefixed $TERPOWER not found in donor bridge')
    c = c.replace(b'\x09$TERPOWER', b'\x0a$TERGROUND', 1)
    c = bytearray(c)
    c[-1] = 0x00
    return bytes(c)

    return [

        {'ref':'R2','value':'2k','left':'N1','right':'N2','x':-2540000,'y':4318000,'role':'upper right vertical in source; horizontal V9 symbol'},

        {'ref':'R6','value':'6k','left':power_node,'right':ground_node,'x':-6350000,'y':-1016000,'role':'bottom horizontal'},
    ]

def topology_r21(power_node='V0', ground_node='G0'):
    r=refs(21); topo=[]; yA=5080000; yB=3556000; yC=2032000; x0=-6350000; dx=2540000
    left=power_node
    for i in range(7):
        right=f'A{i+1}' if i<6 else 'M0'
        topo.append({'ref':r[i],'value':f'{i+1}k','left':left,'right':right,'x':x0+i*dx,'y':yA,'role':'R21 branch A' + ('; powered start' if i==0 else '')})
        left=right
    left=power_node
    for j in range(7):
        idx=7+j; right=f'B{j+1}' if j<6 else 'M0'
        topo.append({'ref':r[idx],'value':f'{idx+1}k','left':left,'right':right,'x':x0+j*dx,'y':yB,'role':'R21 branch B' + ('; powered start' if j==0 else '')})
        left=right
    left='M0'
    for j in range(7):
        idx=14+j; right=f'C{j+1}' if j<6 else ground_node
        topo.append({'ref':r[idx],'value':f'{idx+1}k','left':left,'right':right,'x':x0+j*dx,'y':yC,'role':'R21 tail' + ('; grounded final endpoint' if j==6 else '')})
        left=right
    return topo

def build_resistor_chunk(topology, ground_node: str|None):
    inputs=[]; outputs=[]; groups=[]; maps=[]; ground_count=0
    n=len(topology)
    for idx,m in enumerate(topology, start=1):
        x=m['x']; y=m['y']; left=m['left']; right=m['right']
        in_sym_x=x-508000; out_sym_x=x+1778000
        in_label_x=x-889000; out_label_x=x+2159000
        in_tip_x=x-254000; out_tip_x=x+1524000
        left_pin_x=x; right_pin_x=x+1270000
        inp,in_suf=patch_input(IN_TS[(idx-1)%4], left, in_sym_x, y, in_label_x, y, idx, marker=b'$TERINPUT')
        out_marker=b'$TERGROUND' if (ground_node is not None and right==ground_node) else b'$TEROUTPUT'
        if out_marker == b'$TERGROUND': ground_count += 1
        outp,out_suf=patch_output_groundable(OUT_TS[(idx-1)%4], right, out_sym_x, y, out_label_x, y, idx, marker=out_marker)
        inputs.append(inp); outputs.append(outp)
        res,w1,w2=GROUPS[(idx-1)%4]
        groups.append(patch_res(res, idx, m['ref'], m['value'], x, y, in_suf, out_suf))
        groups.append(patch_wire(w1, in_tip_x, y, left_pin_x, y))
        groups.append(patch_wire(w2, out_tip_x, y, right_pin_x, y))
        mm=dict(m); mm.update({'in_suffix':f'{in_suf:04x}','out_suffix':f'{out_suf:04x}','input_marker':'$TERINPUT','output_marker':out_marker.decode('ascii')})
        maps.append(mm)
    chunk=bytearray(HEADER+b''.join(inputs)+b''.join(outputs)+SEP+b''.join(groups))
    chunk[-1]=0xff
    return bytes(chunk), maps, ground_count

def validate_chunk(c: bytes, n:int, expected_power:int, expected_ground:int, expected_output:int, expected_wire:int):
    issues=[]
    if not c or c[-1] != 0xff: issues.append('final chunk byte not FF')
    if c.count(b'$TERPOWER') != expected_power: issues.append(f'power count {c.count(b"$TERPOWER")} != {expected_power}')
    if c.count(b'$TERGROUND') != expected_ground: issues.append(f'ground count {c.count(b"$TERGROUND")} != {expected_ground}')
    if c.count(b'$TERINPUT') != n: issues.append(f'input count {c.count(b"$TERINPUT")} != {n}')
    if c.count(b'$TEROUTPUT') != expected_output: issues.append(f'output count {c.count(b"$TEROUTPUT")} != {expected_output}')
    if c.count(b'COMPONENT ID') != n: issues.append(f'component count {c.count(b"COMPONENT ID")} != {n}')
    if c.count(b'WIRE') != expected_wire: issues.append(f'wire count {c.count(b"WIRE")} != {expected_wire}')
    return issues

def make_case(case_dir:Path, basename:str, topology, power_node:str, ground_node:str, mode:str, note:str):
    case_dir.mkdir(parents=True, exist_ok=True)
    (case_dir/'CONTROL_E001_EMPTY_BASE.pdsprj').write_bytes(E0_BYTES)
    if mode == 'power_bridge_ground_shortwire':
        bridge = donor_power_bridge(power_node)
        normal, maps, shortwire_ground_count = build_resistor_chunk(topology, ground_node)
        obj = bytearray(bridge + normal[1:])
        expected_power=1; expected_ground=shortwire_ground_count; expected_output=len(topology)-shortwire_ground_count+1; expected_wire=2*len(topology)+1
        method = 'Working power donor bridge plus locked short-wire ground endpoints. The power node remains normal terminal labels in the resistor network; the donor bridge supplies one $TERPOWER connected to a same-name $TEROUTPUT. Ground is direct $TERGROUND short-wire endpoint only.'
    elif mode == 'power_bridge_ground_powerlayout_bridge':
        pbridge = donor_power_bridge(power_node)
        gbridge = donor_ground_bridge_by_power_layout(ground_node)
        normal, maps, shortwire_ground_count = build_resistor_chunk(topology, None)
        obj = bytearray(pbridge + gbridge + normal[1:])
        expected_power=1; expected_ground=1; expected_output=len(topology)+2; expected_wire=2*len(topology)+2
        method = 'Working power donor bridge plus experimental ground bridge made by replacing the power terminal in the same donor power layout with $TERGROUND. Resistor network uses normal terminals for both power and ground labels.'
    else:
        raise ValueError(mode)
    obj[-1]=0xff
    cdb=build_cdb(topology); dsn=build_dsn(bytes(obj))
    proj=case_dir/(basename+'.pdsprj')
    write_project(proj, cdb, dsn)
    (case_dir/(basename+'.ROOT.CDB.bin')).write_bytes(cdb)
    (case_dir/(basename+'.ROOT.DSN.bin')).write_bytes(dsn)
    extracted=extract_chunk_from_dsn(dsn)
    issues=validate_chunk(extracted, len(topology), expected_power, expected_ground, expected_output, expected_wire)
    manifest={
        'case_id':basename,'status':'experimental_unvalidated_terminal_style_power_ground_attempt','mode':mode,'method':method,
        'base':'CONTROL_E001_EMPTY_BASE.pdsprj','donor_file':'New Project(1).pdsprj','donor_sha256':sha256_file(DONOR_PATH),
        'power_node':power_node,'ground_node':ground_node,'note':note,'project_file':proj.name,
        'resistor_count':len(topology),'object_chunk_len':len(obj),'dsn_len':len(dsn),'cdb_len':len(cdb),
        'counts':{'TERPOWER':extracted.count(b'$TERPOWER'),'TERGROUND':extracted.count(b'$TERGROUND'),'TERINPUT':extracted.count(b'$TERINPUT'),'TEROUTPUT':extracted.count(b'$TEROUTPUT'),'WIRE':extracted.count(b'WIRE'),'COMPONENT_ID':extracted.count(b'COMPONENT ID')},
        'static_validation_issues':issues,'topology':maps,
        'hashes':{proj.name:sha256_file(proj),'ROOT.CDB':sha256_bytes(cdb),'ROOT.DSN':sha256_bytes(dsn),'CONTROL_E001_EMPTY_BASE.pdsprj':sha256_bytes(E0_BYTES)},
        'decision_note':'If terminal-style ground bridge fails, keep power bridge + ground short-wire only, or use short-wire for both in production.'}
    (case_dir/'manifest.json').write_text(json.dumps(manifest, indent=2))
    (case_dir/'README_TEST_FIRST.txt').write_text(f'Terminal-style power/ground attempt for {basename}.\n\nOpen:\n1. CONTROL_E001_EMPTY_BASE.pdsprj\n2. {proj.name}\n\nMode: {mode}\n\n{method}\n\nStatic validation issues: {issues}\n')
    shutil.copy(Path(__file__), case_dir/'generation_code_used.py')
    return manifest
